\long\def\splitcols#1#2{%

}

\long\def\lin#1#2{%
	\line{%
		\hsize=\splitsizea
		\vtop{\decora{\leavevmode#1}}%
		\hss
		\hsize=\splitsizeb
		\vtop{\decorb{\leavevmode#2}}%
	}%
	\vskip\splitskip
}
